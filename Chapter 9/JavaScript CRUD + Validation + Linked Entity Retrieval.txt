// Utility: validate fields before CRUD
function validateAccountData(data) {
  if (!data.name || data.name.trim() === "") {
    throw new Error("Account name (text field) is required.");
  }
  if (!data.primarycontactid) {
    throw new Error("Primary Contact (lookup field) is required.");
  }
  if (typeof data.custom_statuscode !== "number") {
    throw new Error("Status (option set) must be a numeric value.");
  }
  if (!data.createdon || isNaN(Date.parse(data.createdon))) {
    throw new Error("CreatedOn (datetime) must be a valid date.");
  }
  if (typeof data.revenue !== "number") {
    throw new Error("Revenue (currency) must be numeric.");
  }
}

// -------------------------------
// CREATE
// -------------------------------
async function createAccount() {
  const token = await getAccessToken();
  const headers = {
    "Authorization": `Bearer ${token}`,
    "Content-Type": "application/json",
    "Accept": "application/json"
  };

  const newAccount = {
    name: "Contoso Ltd",                                // Text
    "primarycontactid@odata.bind": "/contacts(00000000-0000-0000-0000-000000000001)", // Lookup
    custom_statuscode: 100000000,                       // Option set
    createdon: new Date().toISOString(),                // Datetime
    revenue: 50000                                      // Currency
  };

  validateAccountData(newAccount);

  const response = await fetch("https://yourorg.crm.dynamics.com/api/data/v9.2/accounts", {
    method: "POST",
    headers,
    body: JSON.stringify(newAccount)
  });
  const result = await response.json();
  console.log("Created Account ID:", result.id);
  return result.id;
}

async function readAccount(accountId) {
  const token = await getAccessToken();
  const headers = { "Authorization": `Bearer ${token}`, "Accept": "application/json" };

  const response = await fetch(`https://yourorg.crm.dynamics.com/api/data/v9.2/accounts(${accountId})?$select=name,revenue,createdon,custom_statuscode`, { headers });
  const account = await response.json();
  console.log("Account:", account.name, "Revenue:", account.revenue, "Created:", account.createdon);
}

// RETRIEVE MULTIPLE with Linked Entity
// -------------------------------
async function retrieveAccountsWithContacts() {
  const token = await getAccessToken();
  const headers = { "Authorization": `Bearer ${token}`, "Accept": "application/json" };

  const fetchXml = `
    <fetch top="10">
      <entity name="account">
        <attribute name="name" />
        <attribute name="revenue" />
        <attribute name="createdon" />
        <link-entity name="contact" from="contactid" to="primarycontactid" alias="contact">
          <attribute name="firstname" />
          <attribute name="lastname" />
          <attribute name="emailaddress1" />
        </link-entity>
      </entity>
    </fetch>`;

  const encodedFetchXml = encodeURIComponent(fetchXml);
  const response = await fetch(`https://yourorg.crm.dynamics.com/api/data/v9.2/accounts?fetchXml=${encodedFetchXml}`, { headers });
  const results = await response.json();

  results.value.forEach(acc => {
    console.log(`Account: ${acc.name}, Revenue: ${acc.revenue}, Contact: ${acc["contact.firstname"]}, Email: ${acc["contact.emailaddress1"]}`);
  });
}
