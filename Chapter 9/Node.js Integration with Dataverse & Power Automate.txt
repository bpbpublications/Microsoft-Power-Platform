const axios = require('axios');
const qs = require('qs');

const tenantId = "your-tenant-id";
const clientId = "your-client-id";
const clientSecret = "your-client-secret";
const resource = "https://yourorg.crm.dynamics.com";

// 1. Get OAuth token
async function getToken() {
  const tokenUrl = `https://login.microsoftonline.com/${tenantId}/oauth2/token`;
  const body = qs.stringify({
    grant_type: "client_credentials",
    client_id: clientId,
    client_secret: clientSecret,
    resource: resource
  });

  const response = await axios.post(tokenUrl, body);
  return response.data.access_token;
}

// 2. Query Dataverse
async function queryDataverse(token) {
  const url = `${resource}/api/data/v9.2/accounts?$select=name,accountnumber`;
  const response = await axios.get(url, {
    headers: { Authorization: `Bearer ${token}` }
  });
  console.log(response.data.value);
}

// 3. Trigger Power Automate Flow
async function triggerFlow(token) {
  const flowUrl = "https://prod.flow.microsoft.com/api/flows/yourflowid/run";
  const body = { customerId: "12345", status: "active" };

  await axios.post(flowUrl, body, {
    headers: { Authorization: `Bearer ${token}` }
  });
}

(async () => {
  const token = await getToken();
  await queryDataverse(token);
  await triggerFlow(token);
})();
